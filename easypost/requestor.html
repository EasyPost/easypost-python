 40    @classmethod
 41    def _objects_to_ids(cls, param: Dict[str, Any]) -> Dict[str, Any]:
 42        """If providing an object as a parameter to another object,
 43        only pass along the ID so the API will use the object reference correctly."""
 44        if isinstance(param, EasyPostObject):
 45            return {"id": param.id}
 46        elif isinstance(param, dict):
 47            data = {}
 48            for k, v in param.items():
 49                if isinstance(v, list):
 50                    data[k] = [cls._objects_to_ids(item) for item in v]  # type: ignore
 51                else:
 52                    data[k] = cls._objects_to_ids(v)  # type: ignore
 53            return data
 54        else:
 55            return param

 57    @staticmethod
 58    def form_encode_params(
 59        data: Dict, parent_keys: Optional[List[str]] = None, parent_dict: Optional[Dict] = None
 60    ) -> Dict:
 61        """Form-encode a multi-layer dictionary to a one-layer dictionary."""
 62        result = parent_dict or {}
 63        keys = parent_keys or []
 64
 65        for key, value in data.items():
 66            if isinstance(value, dict):
 67                keys.append(key)
 68                result = Requestor.form_encode_params(data=value, parent_keys=keys, parent_dict=result)
 69            else:
 70                dict_key = Requestor._build_dict_key(keys + [key])
 71                result[dict_key] = value
 72        return result

 74    @staticmethod
 75    def _build_dict_key(keys: List[str]) -> str:
 76        """Build a dict key from a list of keys.
 77        Example: [code, number] -> code[number]
 78        """
 79        result = keys[0]
 80
 81        for key in keys[1:]:
 82            result += f"[{key}]"
 83        return result

190    def interpret_response(self, http_body: str, http_status: int) -> Dict[str, Any]:
191        """Interpret the response body we receive from the API."""
192        if http_status == 204:
193            # HTTP 204 does not have any response body and we can just return here
194            return {}
195        try:
196            response = json.loads(http_body)
197        except JSONDecodeError:

199                f"Unable to parse response body from API: {http_body}\nHTTP response code was: {http_status}"

201        if not (200 <= http_status < 300):
202            self.handle_api_error(http_status=http_status, http_body=http_body, response=response)
203        return response

284    def handle_api_error(self, http_status: int, http_body: str, response: Dict[str, Any]) -> None:
285        """Handles API errors returned from EasyPost."""
286        try:
287            error = response["error"]

296    def _utf8(self, value: Union[str, bytes]) -> str:
297        # Python3's str(bytestring) returns "b'bytestring'" so we use .decode instead
298        if isinstance(value, bytes):
299            return value.decode(encoding="utf-8")
300        return value

302    def encode_url_params(self, params: Dict[str, Any], method: RequestMethod) -> Union[str, None]:
303        """Encode params for a URL."""
304        if method not in [RequestMethod.GET, RequestMethod.DELETE]:
305            raise Error("Only GET and DELETE requests support parameters.")
306        converted_params = []
307        for key, value in sorted(params.items()):
308            if value is None:
309                continue  # Don't add Nones to the query
310            elif isinstance(value, datetime.datetime):
311                value = int(time.mktime(value.timetuple()))  # to UTC timestamp
312            converted_params.append((key, value))
313        return urlencode(query=converted_params)
314
315    def add_params_to_url(self, url: str, params: Dict[str, Any], method: RequestMethod) -> str:
316        """Add params to the URL."""
317        if method not in [RequestMethod.GET, RequestMethod.DELETE]:
318            raise Error("Only GET and DELETE requests support parameters.")
319        encoded_params = self.encode_url_params(params=params, method=method)
320        if encoded_params:
321            return "%s?%s" % (url, encoded_params)
322        return url

191    def interpret_response(self, http_body: str, http_status: int) -> Dict[str, Any]:
